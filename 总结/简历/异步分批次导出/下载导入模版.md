@Override
public void downloadTemplate(HttpServletResponse response) {
// 获取业务单位列表
List<String> busUnitValueList = getBusUnitList().stream()
        .map(ComBusUnitTemplateVO::getSetid)
        .collect(Collectors.toList());

// 获取Excel模型信息，包含样例数据
ExcelModelEntity entity = ExportHelper.extractExcelModelPub(ComExecAssistantImportVO.class, null, null, true);

// 设置响应头
String fileName = entity.getFileName();
String file = UriUtils.encode(fileName, StandardCharsets.UTF_8);
response.addHeader("Content-Disposition", "attachment;filename*=utf-8''" + file);
response.addHeader("filename", file);
response.setContentType("application/octet-stream; charset=UTF-8");

// 创建ExcelWriter
ExcelWriter writer;
try {
        writer = new ExcelWriterBuilder()
                .autoCloseStream(true)
                .excelType(ExcelTypeEnum.XLSX)
                .file(response.getOutputStream())
                .build();
} catch (IOException e) {
        log.error("创建ExcelWriter失败", e);
        throw new ServiceException(CommonErrorEnum.EXEC_ASSISTANT_IMPORT_READ_ERROR);
}

// 创建下拉框处理器
List<DropDownBoxWriteHandler.DropDownBoxInfo> dropDownBoxInfos = new ArrayList<>();
if (CollectionUtils.isNotEmpty(busUnitValueList)) {
        DropDownBoxWriteHandler.DropDownBoxInfo dropDownBoxInfo = new DropDownBoxWriteHandler.DropDownBoxInfo();
        dropDownBoxInfo.setFirstCol(0); // 业务单位列，index=0
        dropDownBoxInfo.setLastCol(0);
        dropDownBoxInfo.setFirstRow(2); // 从第3行开始（第1行是备注，第2行是表头，第3行开始是数据）
        dropDownBoxInfo.setEndRow(2000); // 足够大的行数，支持大量数据输入
        dropDownBoxInfo.setValueList(busUnitValueList);
        dropDownBoxInfos.add(dropDownBoxInfo);
}

// 创建WriteSheet，注册下拉框处理器
WriteSheet writeSheet = EasyExcel.writerSheet(entity.getSheetName())
        .registerWriteHandler(new DropDownBoxWriteHandler(dropDownBoxInfos))
        .build();

// 创建备注表头
List<List<String>> headList2 = entity.getHeadList();
List<List<String>> headList1 = new ArrayList<>(headList2.size());
headList2.forEach(h -> headList1.add(Collections.singletonList(entity.getRemark())));

WriteTable table1 = EasyExcel.writerTable(0).head(headList1)
        // 自动合并相同的表头
        .automaticMergeHead(true)
        // 表头、内容样式
        .registerWriteHandler(new ExcelCellStyleStrategy(true))
        // 备注只有一行，表头高度等于内容高度
        .registerWriteHandler(new SimpleRowHeightStyleStrategy((short) entity.getRemarkHeight(), (short) entity.getRemarkHeight()))
        .build();

// 创建数据表头
WriteTable table2 = EasyExcel.writerTable(1).head(headList2)
        // 表头、样例数据样式
        .registerWriteHandler(new ExcelCellStyleStrategy(entity.getAlignmentList(), true))
        // 设置列宽
        .registerWriteHandler(new WidthStyleStrategy(entity.getWidthList()))
        // 设置表头和内容高度
        .registerWriteHandler(new SimpleRowHeightStyleStrategy((short) entity.getTitleHeight(), (short) entity.getContentHeight()))
        .build();

// 写入备注行
writer.write(Collections.emptyList(), writeSheet, table1);

// 写入样例数据
writer.write(entity.getExampleList(), writeSheet, table2);

// 完成写入
writer.finish();
}
